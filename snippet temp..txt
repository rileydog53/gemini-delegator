Here is the plan and the subsequent code.
1. Architectural Plan
Design Pattern: Model-View-Controller (MVC) / Event-Driven
	•	View (GUI): Built with PyQt6 using a clean QTabWidget structure (perfect for a native Mac look).
	◦	Top Bar: Global settings like the Model Selector (mocked with a few of the 49 models) and a status bar.
	◦	Tabs: Research, Code, and Image. Each encapsulates its specific inputs and options.
	•	Controller (Threaded Workers): API calls to Gemini/Imagen can take seconds to minutes. I implemented a QThread worker class (GeminiWorker). It takes task definitions, executes them in the background, and communicates back to the main thread via Qt Signals (finished, error).
	•	Model/State (Mock Backend): For the GUI evaluation script below, I built a mock backend that simulates network latency, rate-limit handling, the 2000-character UI truncation, and the auto-saving mechanism to outputs/ and the Desktop.
2. Component Breakdown
	•	Global Options: QComboBox for Model Selection (Gemini 1.5 Pro, Gemma, Veo, etc.).
	•	Research Tab: QTextEdit for the query, QComboBox for depth (Basic, Intermediate, Advanced, Expert). Read-only text area for truncated JSON preview.
	•	Code Tab: Same layout as Research, but uses a monospace font (Courier or Menlo on Mac) for the output preview.
	•	Image Tab: Rich parameter inputs utilizing QFormLayout. Includes a QSpinBox (1–4 images), and multiple QComboBoxes for Aspect Ratio, Style, Lighting, Mood, and Quality. Includes a QLineEdit for Negative Prompts.
	•	Auto-Save Layer: A utility function attached to the worker's completion signal that writes the raw payload to the filesystem before truncating it for the UI.
3. Complete PyQt6 Implementation
You will need to install PyQt6 (pip install PyQt6). You can copy, paste, and run this code directly. It simulates the backend so you can evaluate the UX, threading, and layout immediately without needing your actual API keys.

# ==========================================
# WORKER THREAD (Handles Async API Calls)
# ==========================================
class GeminiWorker(QThread):
    finished = pyqtSignal(dict)
    error = pyqtSignal(str)

    def __init__(self, task_type, payload):
        super().__init__()
        self.task_type = task_type
        self.payload = payload

    def run(self):
        try:
            # Simulate API Latency, thinking tokens, and rate limits
            time.sleep(2) 
            
            # Simulated Response Generation
            if self.task_type in ["Research", "Code"]:
                content = f"Simulated {self.task_type} response at {self.payload['depth']} depth.\n"
                # Add padding to simulate a long response (testing the 2000 char cap)
                content += "Lorem ipsum dolor sit amet, " * 100 
                
                response = {
                    "status": "success",
                    "task_type": self.task_type,
                    "model": self.payload["model"],
                    "data": {
                        "findings" if self.task_type == "Research" else "code": content,
                        "sources": ["https://google.com/search?q=gemini"],
                        "thinking_tokens_used": 142
                    }
                }
            elif self.task_type == "Image":
                response = {
                    "status": "success",
                    "task_type": self.task_type,
                    "model": self.payload["model"],
                    "data": {
                        "images_generated": self.payload["count"],
                        "paths": ["mock_image_path.png"] * self.payload["count"]
                    }
                }
            self.finished.emit(response)
        except Exception as e:
            self.error.emit(str(e))

    # ==========================================
    # EXECUTION TRIGGERS
    # ==========================================
    def run_research(self):
        query = self.research_query.toPlainText().strip()
        if not query: return
        self.research_output.clear()
        
        payload = {
            "model": self.model_combo.currentText(),
            "query": query,
            "depth": self.research_depth.currentText()
        }
        self.start_worker("Research", payload)

    def run_code(self):
        query = self.code_query.toPlainText().strip()
        if not query: return
        self.code_output.clear()
        
        payload = {
            "model": self.model_combo.currentText(),
            "query": query,
            "depth": self.code_depth.currentText()
        }
        self.start_worker("Code", payload)

    def run_image(self):
        prompt = self.image_prompt.toPlainText().strip()
        if not prompt: return
        self.image_preview_label.setText("Generating...")
        
        payload = {
            "model": self.model_combo.currentText(),
            "prompt": prompt,
            "neg_prompt": self.image_neg_prompt.text(),
            "count": self.img_count.value(),
            "aspect_ratio": self.img_ar.currentText(),
            "style": self.img_style.currentText(),
            "lighting": self.img_lighting.currentText(),
            "mood": self.img_mood.currentText(),
            "quality": self.img_quality.currentText()
        }
        self.start_worker("Image", payload)

    def start_worker(self, task_type, payload):
        self.status_label.setText(f"Status: Generating {task_type}...")
        self.tabs.setEnabled(False) # Prevent multi-spam
        
        self.worker = GeminiWorker(task_type, payload)
        self.worker.finished.connect(self.handle_worker_success)
        self.worker.error.connect(self.handle_worker_error)
        self.worker.start()

    # ==========================================
    # RESPONSE HANDLING & AUTO-SAVE
    # ==========================================
    def handle_worker_success(self, response):
        self.tabs.setEnabled(True)
        self.status_label.setText("Status: Idle")
        task_type = response.get("task_type")
        
        # 1. AUTO SAVE LOGIC
        timestamp = int(time.time())
        filename = f"{task_type.lower()}_{timestamp}.json"
        
        # Save to outputs/
        with open(self.outputs_dir / filename, 'w') as f:
            json.dump(response, f, indent=4)
        
        # Save to Desktop
        with open(self.desktop_dir / filename, 'w') as f:
            json.dump(response, f, indent=4)

        # 2. UI DISPLAY (Cap at 2000 chars)
        display_text = json.dumps(response, indent=2)
        if len(display_text) > 2000:
            display_text = display_text[:2000] + "\n\n... [TRUNCATED FOR READABILITY. FULL FILE SAVED TO DESKTOP/OUTPUTS]"

        if task_type == "Research":
            self.research_output.setPlainText(display_text)
        elif task_type == "Code":
            self.code_output.setPlainText(display_text)
        elif task_type == "Image":
            count = response["data"]["images_generated"]
            self.image_preview_label.setText(f"Successfully generated {count} image(s).\n\n" + display_text)
            
            # NOTE: In production, load actual paths via QPixmap
            # pixmap = QPixmap(actual_image_path)
            # self.image_preview_label.setPixmap(pixmap.scaled(self.image_preview_label.size(), Qt.AspectRatioMode.KeepAspectRatio))

    def handle_worker_error(self, error_msg):
        self.tabs.setEnabled(True)
        self.status_label.setText("Status: Error")
        QMessageBox.critical(self, "API Error", f"An error occurred:\n{error_msg}")
